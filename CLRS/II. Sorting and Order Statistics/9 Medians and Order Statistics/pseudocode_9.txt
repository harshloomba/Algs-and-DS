MINIMUM(A)
1. min = A[1]
2. for i = 2 to A.length
3.	if min > A[i]
4.		min = A[i]
5. return min

MAXIMUM(A)
1. max = A[1]
2. for i = 2 to A.length
3.	if max < A[i]
4.		max = A[i]
5. return max

RANDOMIZED-SELECT(A,p,r,i)
1. if p == r
2.	return A[p]
3. q = RANDOMIZED-PARTITION(A,p,r)
4. k = q - p + 1
5. if i == k	// the pivot is the answer
6.	return A[q]
7. else if i < k
8.	return RANDOMIZED-SELECT(A, p, q - 1, i)
9. else return RANDOMIZED-SELECT(A, q + 1, r, i - k)
